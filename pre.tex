\begin{frame}{Executive Summary}
    \small
    \begin{itemize}
        \item This project builds an integrated 2D Boeing 747 circular-tracking workflow for control, simulation, and estimation.
        \item The framework combines LQR inner-loop regulation, RK4 nonlinear state propagation, and Gaussian Bayes/MAP state estimation.
        \item Under the baseline setting, the aircraft converges smoothly to the reference circle with near-zero steady-state cross-track error and tight speed regulation.
        \item The main message is that the proposed pipeline is accurate, computationally light, and robust enough for coursework-scale flight simulation.
    \end{itemize}
\end{frame}

\section{Problem}

\begin{frame}{Problem Formulation}
    \small
    \[
        \mathbf{x} = [x,\, y,\, \psi,\, \phi,\, v]^\top,
        \qquad
        \mathbf{u} = [\phi_{\mathrm{cmd}},\, T]^\top
    \]

    \begin{itemize}
        \item Objective: drive a Boeing 747 model to track a counterclockwise circular reference while maintaining the target speed.
        \item Reference geometry: circle centered at $(0,0)$ with $R = 120{,}000\,\mathrm{m}$ and $v_{\mathrm{ref}} = 210\,\mathrm{m/s}$.
        \item Performance goals: reduce cross-track error, keep heading and bank response smooth, respect bank/thrust limits, and regulate speed.
        \item Assumptions: planar flight, quasi-steady drag, discrete control update, and Gaussian measurement noise.
    \end{itemize}
\end{frame}

\begin{frame}{Why This Workflow Matters}
    \small
    \begin{itemize}
        \item The project connects optimization, differential-equation solving, and uncertainty-aware estimation in one aerospace workflow.
        \item LQR offers predictable tuning and stable heading-bank regulation for circular tracking.
        \item RK4 improves nonlinear propagation accuracy compared with low-order integration.
        \item Bayesian estimation quantifies innovation, posterior error, and covariance contraction under noisy measurements.
    \end{itemize}
\end{frame}

\begin{frame}{Modeling and Algorithms}
    \scriptsize
    \begin{block}{Nonlinear Planar Dynamics}
        The five-state model propagates position, heading, bank angle, and speed. The heading rate follows $\dot{\psi} = \frac{g \tan \phi}{v}$, while bank angle follows a first-order lag and speed is driven by thrust minus aerodynamic drag.
    \end{block}

    \begin{block}{Control, Integration, and Estimation}
        \setlength{\itemsep}{0.2em}
        \begin{itemize}
            \item LQR stabilizes heading and bank-angle errors around the circular-tracking reference.
            \item RK4 propagates the nonlinear dynamics at $\Delta t = 0.05\,\mathrm{s}$ over an $800\,\mathrm{s}$ horizon.
            \item A Gaussian Bayes/MAP estimator processes noisy measurements and reports estimation consistency.
        \end{itemize}
    \end{block}

    \begin{alertblock}{Baseline Setup}
        The aircraft starts $2\,\mathrm{km}$ outside the target circle, and the evaluation tracks error, speed, actuator usage, runtime, and estimation consistency.
    \end{alertblock}
\end{frame}

\section{Results}

\begin{frame}{Nominal Closed-Loop Performance}

        \begin{column}{0.58\textwidth}
            \scriptsize

            \begin{tabular}{|p{0.60\linewidth}|p{0.26\linewidth}|}
                \hline
                \textbf{Metric} & \textbf{Value} \\
                \hline
                Final cross-track error & $0.218\,\mathrm{m}$ \\
                \hline
                Cross-track RMSE after transient & $5.56\,\mathrm{m}$ \\
                \hline
                Settling time to $|e_{ct}|<100\,\mathrm{m}$ & $78.35\,\mathrm{s}$ \\
                \hline
                Heading error RMSE & $2.31^\circ$ \\
                \hline
                Max bank angle & $27.67^\circ$ \\
                \hline
                Speed RMSE to $210\,\mathrm{m/s}$ & $0.276\,\mathrm{m/s}$ \\
                \hline
                Control / estimator time & \shortstack{$0.0165\,\mathrm{ms}$\\$0.0344\,\mathrm{ms}$} \\
                \hline
            \end{tabular}
        \end{column}
        \begin{column}{0.42\textwidth}
            \small
            \begin{itemize}
                \item The controller removes the initial offset quickly and reaches very small steady-state error.
                \item Bank angle stays inside the $35^\circ$ envelope, and saturation is rare.
                \item Speed remains tightly clustered around the $210\,\mathrm{m/s}$ reference.
                \item Runtime is negligible compared with the $50\,\mathrm{ms}$ control interval.
            \end{itemize}
        \end{column}
    \end{columns}

            \setlength{\itemsep}{0.45em}
            \begin{itemize}
                \item Aggressive LQR tuning improves tracking but increases peak bank demand and saturation risk.

\section{Conclusions}

\begin{frame}{Discussion and Future Work}
    \small
    \begin{itemize}
        \item Strengths: stable circular tracking, low computational cost, clear controller-estimator architecture, and interpretable uncertainty diagnostics.
        \item Main limitations: planar flight assumption, simplified actuator and disturbance modeling, first-order discretization in LQR design, and true-state feedback instead of estimated-state feedback.
        \item Future work: close the loop around estimated states, add wind/disturbances, compare with MPC or nonlinear guidance laws, and extend toward higher-fidelity aircraft dynamics.
    \end{itemize}
\end{frame}
