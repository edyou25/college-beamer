\section{Overview}

\begin{frame}{Executive Summary}
    \small
    \begin{itemize}
        \item This project builds an integrated 2D Boeing 747 circular-tracking workflow for control, simulation, and estimation.
        \item The framework combines LQR inner-loop regulation, RK4 nonlinear state propagation, and Gaussian Bayes/MAP state estimation.
        \item Under the baseline setting, the aircraft converges smoothly to the reference circle with near-zero steady-state cross-track error and tight speed regulation.
        \item The main message is that the proposed pipeline is accurate, computationally light, and presentation-ready for coursework-scale flight simulation.
    \end{itemize}
\end{frame}

\begin{frame}{Project Scope and Baseline Setup}
    \begin{columns}[T]
        \begin{column}{0.43\textwidth}
            \small
            \[
                \mathbf{x} = [x,\, y,\, \psi,\, \phi,\, v]^\top,
                \qquad
                \mathbf{u} = [\phi_{\mathrm{cmd}},\, T]^\top
            \]

            \begin{itemize}
                \item Objective: converge from an off-circle initial condition to a counterclockwise reference orbit.
                \item Controller stack: geometry guidance, LQR heading-bank regulation, and thrust hold.
                \item Evaluation: tracking accuracy, actuator usage, runtime, and estimation consistency.
            \end{itemize}
        \end{column}
        \begin{column}{0.57\textwidth}

                \hline
                Reference & $R=120\,\mathrm{km}$, $v_{\mathrm{ref}}=210\,\mathrm{m/s}$ \\
                \hline
                Initial state & $[R+2000,\,0,\,96^\circ,\,0^\circ,\,210]$ \\
                \hline
                Simulation & $\Delta t=0.05\,\mathrm{s}$, $t_{end}=800\,\mathrm{s}$ \\
                \hline
                Constraints & $|\phi|\le 35^\circ$, clipped thrust and speed envelope \\
                \hline
                Noise model & Gaussian measurements for Bayes/MAP diagnostics \\
                \hline
            \end{tabular}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Why This Workflow Matters}
    \small
    \begin{itemize}
        \item The project connects optimization, differential-equation solving, and uncertainty-aware estimation in one aerospace workflow.
        \item LQR offers predictable tuning and stable heading-bank regulation for circular tracking.
        \item RK4 improves nonlinear propagation accuracy compared with low-order integration.
        \item Bayesian estimation quantifies innovation, posterior error, and covariance contraction under noisy measurements.
    \end{itemize}
\end{frame}

\begin{frame}{Guidance and Control Logic}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \centering
            \includegraphics[width=\linewidth,height=0.99\textheight,keepaspectratio]{latex/assets/controller_logic.png}
        \end{column}
        \begin{column}{0.52\textwidth}
            \small
            \begin{itemize}
                \item The geometry block computes current radius, tangent heading, and signed cross-track error.
                \item Desired heading is corrected by a bounded nonlinear term based on $e_{ct}$.
                \item Feedforward bank comes from circular-motion curvature, while LQR regulates $e_\psi$ and $e_\phi$.
                \item A separate thrust channel keeps the aircraft close to the reference speed.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}[fragile]{Simulation Loop Sketch}
\begin{lstlisting}[language=Python,basicstyle=\ttfamily\tiny,breaklines=true,columns=fullflexible,showstringspaces=false]
for k in range(N):
    r_now, psi_tangent, e_ct = geometry(state, R)
    psi_des = psi_tangent + clip(arctan(k_ct * (-e_ct)), deg=25)
    phi_ff = atan(v_ref**2 / (g * R))
    e = [wrap(psi - psi_des), phi - phi_ff]
    delta_phi = -K @ e
    phi_cmd = clip(phi_ff + delta_phi, -phi_max, phi_max)
    thrust = D_guess + k_v * (v_ref - v)
    state = rk4_step(f, state, [phi_cmd, thrust], dt)
    z = state + measurement_noise()
    x_hat, P = gaussian_map_update(x_hat, P, z)
\end{lstlisting}
\end{frame}

\section{Results}

\begin{frame}{Nominal Closed-Loop Performance}
    \begin{columns}[T]
        \begin{column}{0.58\textwidth}
            \scriptsize
            \renewcommand{\arraystretch}{1.12}
            \begin{tabular}{|p{0.60\linewidth}|p{0.26\linewidth}|}
                \hline
                \textbf{Metric} & \textbf{Value} \\
                \hline
                Final cross-track error & $0.218\,\mathrm{m}$ \\
                \hline

                \hline

                \hline

                \hline
                Max bank angle & $27.67^\circ$ \\
                \hline
                Speed RMSE to $210\,\mathrm{m/s}$ & $0.276\,\mathrm{m/s}$ \\
                \hline
                Control / estimator time & \shortstack{$0.0165\,\mathrm{ms}$\\$0.0344\,\mathrm{ms}$} \\
                \hline
            \end{tabular}
        \end{column}
        \begin{column}{0.42\textwidth}
            \small
            \begin{itemize}
                \item The controller removes the initial offset quickly and reaches very small steady-state error.
                \item Bank angle stays inside the $35^\circ$ envelope, and saturation is rare.
                \item Speed remains tightly clustered around the $210\,\mathrm{m/s}$ reference.
                \item Runtime is negligible compared with the $50\,\mathrm{ms}$ control interval.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Key Findings and Future Work}
    \small
    \begin{itemize}
        \item Stable circular tracking, low computational cost, and interpretable uncertainty diagnostics are the main strengths of the current pipeline.
        \item The largest modeling limits are planar flight, simplified actuator and disturbance models, first-order LQR discretization, and true-state feedback instead of estimated-state feedback.
        \item Natural next steps are estimated-state closed-loop control, wind and disturbance injection, and comparison with MPC or nonlinear guidance laws.
    \end{itemize}
\end{frame}
